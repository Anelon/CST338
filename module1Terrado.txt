 *This program is designed to have two parts.
 * Part one takes a user's first and last name from console and places them in their own separate variable.
 * The program then concatenates the two names with a space between names and stores it into fullName.
 * The length of fullName is stored into nameLength
 * The program prints the fullName and the length of fullName (including space)
 * Lastly, part one prints fullName in all upper case and all lower case
 * 
 * Part two requests the user's input for hours in a range of 12 to 20, with 3 decimal points of accuracy
 * It then rounds to 1 decimal point
 */

import java.util.Scanner;
public class moduleOne {
   
   public static void main(String[] args) {
   Scanner userInput = new Scanner(System.in);
   
   //Part 1
   
   //Takes user input, places input in firstName and lastName
   System.out.println("Please enter your first name, with the first letter capitalized.");
   String firstName = userInput.nextLine();
   System.out.println("Please enter your last name, with the first letter capitalized.");
   String lastName = userInput.nextLine();
   
   //Concatenates names and finds length 
   String fullName = (firstName+ " " + lastName);
   int nameLength = fullName.length();
   
   //Prints fullName, length of fullName, and fullName in upper/lower case
   System.out.println("Your full name is: " + fullName);
   System.out.println("Your name's length (including space) is: " + (nameLength));
   System.out.println("Your name in upper case: " + fullName.toUpperCase());
   System.out.println("Your name in lower case: " + fullName.toLowerCase());

   //Part 2
   
   //Static variables
   int MIN_HOURS = 12;
   int MAX_HOURS = 20;
   
   //Requests user input and places input in variable hours
   System.out.println("Enter study hours, from a range of " + MIN_HOURS + " to " + MAX_HOURS + ", with 3 decimal places of accuracy");
   double hours = userInput.nextDouble();
   
   //Prints user input, to one decimal place
   System.out.println("Your adjusted hours are: ");
   System.out.printf("%.1f", hours);

   }

}
